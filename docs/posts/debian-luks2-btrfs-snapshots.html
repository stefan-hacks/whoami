Debian with LUKS2, BTRFS & Snapper: Production Installation Guide

Terminal-based walkthrough: From Debian net-installer to production-ready GNOME system with encrypted BTRFS, automatic snapshots, and bootable recovery.

Goal: Install Debian with LUKS2 encryption, BTRFS with Kali "Unkaputtbar" subvolumes, separate unencrypted /boot, Snapper for automatic APT snapshots, and full GNOME desktop.

Architecture

nvme0n1p1  1GB   FAT32   /boot/efi     (ESP - UEFI bootloader)
nvme0n1p2  2GB   ext4    /boot         (kernels, initramfs - unencrypted)
nvme0n1p3  Rest  LUKS2   crypt_root    (encrypted BTRFS container)
           └── @           (root filesystem)
           ├── @home       (/home)
           ├── @root       (/root)
           ├── @srv        (/srv)
           ├── @tmp        (/tmp)
           ├── @usr_local  (/usr/local)
           ├── @var_log    (/var/log)
           └── @.snapshots (snapper snapshots)

Phase 1: Boot Installer & Enter Terminal

Boot from Debian net-install ISO. At the first installer screen:

  1. Press Tab to edit boot parameters
  2. Add console=tty1 if needed
  3. Press Enter to boot

When the installer loads (language/location screens), switch to terminal:

# Press: Ctrl + Alt + F2
# Then press Enter to activate the terminal

Phase 2: Partition the Disk

# Identify your target disk
lsblk -dpno NAME,SIZE,MODEL | grep -E 'nvme|sd|vd'

# Set disk variable (adjust as needed)
DISK="/dev/nvme0n1"

# Create GPT partition table and partitions
sgdisk --zap-all "$DISK"
sgdisk -n "1:0:+1G" -t "1:ef00" -c "1:ESP" "$DISK"
sgdisk -n "2:0:+2G" -t "2:8300" -c "2:BOOT" "$DISK"
sgdisk -n "3:0:0" -t "3:8300" -c "3:CRYPT" "$DISK"
lsblk "$DISK"

Phase 3: Format Partitions

# Format ESP
mkfs.fat -F32 -n ESP "${DISK}p1"

# Format /boot
mkfs.ext4 -L BOOT "${DISK}p2"
tune2fs -m 0 "${DISK}p2"

# Setup LUKS2 on root partition
cryptsetup luksFormat --type luks2 "${DISK}p3"
cryptsetup open "${DISK}p3" crypt_root

# Create BTRFS filesystem
mkfs.btrfs -L DEBIAN /dev/mapper/crypt_root

Phase 4: Create BTRFS Subvolumes

# Mount the BTRFS root
mount /dev/mapper/crypt_root /mnt

# Create Kali "Unkaputtbar" subvolumes
btrfs subvolume create /mnt/@
btrfs subvolume create /mnt/@home
btrfs subvolume create /mnt/@root
btrfs subvolume create /mnt/@srv
btrfs subvolume create /mnt/@tmp
btrfs subvolume create /mnt/@usr_local
btrfs subvolume create /mnt/@var_log
btrfs subvolume create /mnt/@.snapshots

# Unmount root
umount /mnt

Phase 5: Mount All Filesystems

# Mount subvolumes with optimal flags
mount -o noatime,compress=zstd:1,subvol=@ /dev/mapper/crypt_root /mnt
mkdir -p /mnt/{boot,home,root,srv,tmp,usr/local,var/log,.snapshots}

mount -o noatime,compress=zstd:1,subvol=@home /dev/mapper/crypt_root /mnt/home
mount -o noatime,compress=zstd:1,subvol=@root /dev/mapper/crypt_root /mnt/root
mount -o noatime,compress=zstd:1,subvol=@srv /dev/mapper/crypt_root /mnt/srv
mount -o noatime,nodiratime,subvol=@tmp /dev/mapper/crypt_root /mnt/tmp
mount -o noatime,compress=zstd:1,subvol=@usr_local /dev/mapper/crypt_root /mnt/usr/local
mount -o noatime,subvol=@var_log /dev/mapper/crypt_root /mnt/var/log
mount -o noatime,subvol=@.snapshots /dev/mapper/crypt_root /mnt/.snapshots

# Mount boot and ESP
mount "${DISK}p2" /mnt/boot
mkdir -p /mnt/boot/efi
mount "${DISK}p1" /mnt/boot/efi

# Verify layout
lsblk "$DISK"

Phase 6: Bootstrap Debian with debootstrap

# Install debootstrap
debootstrap --version || apt update && apt install -y debootstrap

# Bootstrap testing base system
debootstrap testing /mnt http://deb.debian.org/debian

# Mount pseudo-filesystems for chroot
mount --make-rslave --rbind /dev /mnt/dev
mount --make-rslave --rbind /sys /mnt/sys
mount --make-rslave --rbind /proc /mnt/proc
mount --make-rslave --rbind /run /mnt/run

Phase 7: Enter Chroot & Configure Base System

# Enter chroot
chroot /mnt /bin/bash

# Set hostname
echo "your-hostname" > /etc/hostname

# Configure hosts
cat > /etc/hosts << 'EOF'
127.0.0.1   localhost
127.0.1.1   your-hostname
::1         localhost ip6-localhost ip6-loopback
ff02::1     ip6-allnodes
ff02::2     ip6-allrouters
EOF

# Set root password
passwd

# Add user
useradd -m -s /bin/bash -G sudo your-username
passwd your-username

# Configure APT sources
cat > /etc/apt/sources.list << 'EOF'
deb http://deb.debian.org/debian testing main contrib non-free non-free-firmware
deb http://deb.debian.org/debian testing-updates main contrib non-free non-free-firmware
deb http://security.debian.org/debian-security testing-security main contrib non-free non-free-firmware
EOF

apt update

Phase 8: Install Kernel & Essential Packages

# Install kernel, firmware, and essentials
apt install -y linux-headers-amd64 linux-image-amd64 firmware-linux \
    firmware-linux-nonfree firmware-misc-nonfree

# Install networking, crypto, and base tools
apt install -y network-manager wireless-tools wpasupplicant \
    cryptsetup-initramfs btrfs-progs grub-efi-amd64 sudo

# Install GNOME desktop
apt install -y task-gnome-desktop gdm3

Phase 9: Configure fstab

# Get UUIDs
crypt_uuid=$(blkid -s UUID -o value "${DISK}p3")
boot_uuid=$(blkid -s UUID -o value "${DISK}p2")
esp_uuid=$(blkid -s UUID -o value "${DISK}p1")

# Create fstab
cat > /etc/fstab << EOF
# /etc/fstab - LUKS2 BTRFS with Kali Unkaputtbar subvolumes
#                                                                      

# Encrypted BTRFS subvolumes
UUID=${crypt_uuid}  /              btrfs   noatime,compress=zstd:1,subvol=@,discard=async                          0 1
UUID=${crypt_uuid}  /home          btrfs   noatime,compress=zstd:1,subvol=@home,discard=async                    0 2
UUID=${crypt_uuid}  /root          btrfs   noatime,compress=zstd:1,subvol=@root,discard=async                    0 2
UUID=${crypt_uuid}  /srv           btrfs   noatime,compress=zstd:1,subvol=@srv,discard=async                     0 2
UUID=${crypt_uuid}  /tmp           btrfs   noatime,nodiratime,subvol=@tmp,discard=async                        0 2
UUID=${crypt_uuid}  /usr/local     btrfs   noatime,compress=zstd:1,subvol=@usr_local,discard=async               0 2
UUID=${crypt_uuid}  /var/log       btrfs   noatime,subvol=@var_log,discard=async                               0 2
UUID=${crypt_uuid}  /.snapshots    btrfs   noatime,subvol=@.snapshots,discard=async                          0 2

# /boot (unencrypted ext4)
UUID=${boot_uuid}   /boot          ext4    defaults,noatime                                                        0 2

# EFI System Partition
UUID=${esp_uuid}    /boot/efi      vfat    umask=0077,noatime                                                    0 1
EOF
cat /etc/fstab

Phase 10: Configure Crypttab & Initramfs

# Configure crypttab
echo "crypt_root UUID=${crypt_uuid} none luks,discard" > /etc/crypttab
cat /etc/crypttab

# Set CPU microcode (for faster boot)
apt install -y amd64-microcode || apt install -y intel-microcode

# Generate initramfs
update-initramfs -u -k all

Phase 11: Install GRUB Bootloader

# Install GRUB to ESP
grub-install --target=x86_64-efi --efi-directory=/boot/efi --bootloader-id=debian --recheck

# Update GRUB configuration
cat > /etc/default/grub << 'EOF'
GRUB_DEFAULT=0
GRUB_TIMEOUT=5
GRUB_DISTRIBUTOR='Debian'
GRUB_CMDLINE_LINUX_DEFAULT="quiet"
GRUB_CMDLINE_LINUX="cryptdevice=UUID=PLACEHOLDER:crypt_root"
GRUB_TERMINAL=console
GRUB_GFXMODE=1024x768
EOF

# Replace placeholder with actual UUID
sed -i "s/PLACEHOLDER/${crypt_uuid}/g" /etc/default/grub

# Generate grub.cfg
update-grub

Phase 12: Install Snapper & Configure

# Install Snapper
apt install -y snapper snapper-gui

# Create snapper config for root
snapper --config root create-config /

# Configure snapper for automatic APT snapshots
cat > /etc/snapper/configs/root << 'EOF'
# snapper configuration for root filesystem
SUBVOLUME="/"
FSTYPE="btrfs"
QGROUP=""
SPACE_LIMIT="0.5"
FREE_LIMIT="0.2"

# Snapshot types
TIMELINE_CREATE="no"
NUMBER_CLEANUP="yes"
NUMBER_MIN_AGE="1800"
NUMBER_LIMIT="10"
TIMELINE_CLEANUP="no"
EMPTY_PRE_POST_CLEANUP="yes"
EOF

# Enable APT hooks for automatic snapshots
apt install -y python3-dbus

# Create pre-apt hook for snapshot creation
cat > /etc/apt/apt.conf.d/80-snapper-pre << 'EOF'
// Create snapshot before APT operations
DPkg::Pre-Invoke {"if which snapper >/dev/null 2>&1; then snapper --config root create --description \"apt-pre-$(date +%Y%m%d-%H%M%S)\" --cleanup-algorithm number 2>/dev/null || true; fi"; };
EOF

# Create post-apt hook for snapshot creation
cat > /etc/apt/apt.conf.d/80-snapper-post << 'EOF'
// Create snapshot after APT operations
DPkg::Post-Invoke {"if which snapper >/dev/null 2>&1; then snapper --config root create --description \"apt-post-$(date +%Y%m%d-%H%M%S)\" --cleanup-algorithm number 2>/dev/null || true; fi"; };
EOF

Phase 13: Install grub-btrfs (Bootable Snapshots)

# Install dependencies
apt install -y git build-essential gettext

# Clone and install grub-btrfs
cd /tmp
git clone https://github.com/Antynea/grub-btrfs.git
cd grub-btrfs
make install

# Update GRUB after snapshots exist
update-grub

Phase 14: Exit & Reboot

# Exit chroot
exit

# Unmount everything in reverse order
umount -R /mnt/dev
umount -R /mnt/sys
umount -R /mnt/proc
umount -R /mnt/run
umount /mnt/boot/efi
umount /mnt/boot
umount -R /mnt

# Close LUKS
cryptsetup close crypt_root

# Reboot
reboot

Working with Snapshots

# List snapshots
snapper --config root list

# Create manual snapshot
snapper --config root create --description "Before big change"

# Delete old snapshots
snapper --config root delete 3

# Compare snapshots
snapper --config root diff 1..2

# Rollback to snapshot (reboot after)
snapper --config root rollback 3

Boot from Snapshot (Recovery)

  1. Reboot and hold Shift to show GRUB menu
  2. Select Debian Snapshots submenu
  3. Choose any snapshot to boot read-write
  4. Test the system - if satisfied, run: snapper rollback <number>

Verification Commands

# Check subvolumes
btrfs subvolume list / | sort

# Check filesystem compression
btrfs filesystem df /

# Check encryption status
cryptsetup status crypt_root

# Check snapper config
snapper get-config

# Check GRUB detection
grub-editenv list
Installation Complete! You now have a production-ready Debian system with:
  • LUKS2 encryption for data protection
  • Separate unencrypted /boot for reliability
  • BTRFS with Kali "Unkaputtbar" subvolumes
  • Snapper auto-snapshots on every APT operation
  • Bootable snapshots via grub-btrfs menu
  • Full GNOME desktop environment
